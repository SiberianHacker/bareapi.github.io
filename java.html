Java код для использования API, отправка капчи из файла или из буфера

									import java.awt.image.BufferedImage;
import java.io.*;
import java.net.HttpURLConnection;
import java.net.URL;
import java.net.URLEncoder;
import java.nio.file.Files;
import java.nio.file.Paths;
import java.util.Base64;
import java.util.Scanner;

import javax.imageio.ImageIO;

public class CaptchaSolver {

    private static final String API_KEY = "Ваш ключ Bare API";
    private static final String SITE = "http://5.42.211.111/";
    private static final int TIMEOUT = 10000;
    private static final long GET_DELAY = 800;

    public static String imageToBase64(String imagePath) throws IOException {
        byte[] imageBytes = Files.readAllBytes(Paths.get(imagePath));
        return Base64.getEncoder().encodeToString(imageBytes);
    }
    
    public static String imageToBase64(BufferedImage image) throws IOException {
	    ByteArrayOutputStream byteArrayOutputStream = new ByteArrayOutputStream();
	    ImageIO.write(image, "png", byteArrayOutputStream);
	    return Base64.getEncoder().encodeToString(byteArrayOutputStream.toByteArray());
    }
    
    public static String sendPost(String url, String params) throws IOException {
        @SuppressWarnings("deprecation")
		URL obj = new URL(url);
        HttpURLConnection connection = (HttpURLConnection) obj.openConnection();

        connection.setRequestMethod("POST");
        connection.setConnectTimeout(TIMEOUT);
        connection.setReadTimeout(TIMEOUT);
        connection.setDoOutput(true);

        try (OutputStream os = connection.getOutputStream()) {
            byte[] input = params.getBytes("UTF-8");
            os.write(input, 0, input.length);
        }

        BufferedReader reader = new BufferedReader(new InputStreamReader(connection.getInputStream(), "UTF-8"));
        StringBuilder response = new StringBuilder();
        String line;

        while ((line = reader.readLine()) != null) {
            response.append(line);
        }

        reader.close();
        return response.toString();
    }

    public static String sendGet(String url) throws IOException {
        @SuppressWarnings("deprecation")
		URL obj = new URL(url);
        HttpURLConnection connection = (HttpURLConnection) obj.openConnection();

        connection.setRequestMethod("GET");
        connection.setConnectTimeout(TIMEOUT);
        connection.setReadTimeout(TIMEOUT);

        BufferedReader reader = new BufferedReader(new InputStreamReader(connection.getInputStream(), "UTF-8"));

        StringBuilder response = new StringBuilder();
        String line;

        while ((line = reader.readLine()) != null) {
            response.append(line);
        }

        reader.close();
        return response.toString();
    }

    public static String solveCaptcha(String filename) {
        try {
            String base64Image = imageToBase64(filename);
            String postParams = "key=" + URLEncoder.encode(API_KEY, "UTF-8") + "&method=base64" + "&body=" + URLEncoder.encode(base64Image, "UTF-8");
            String postResponse = sendPost(SITE + "in.php", postParams);
            
            if (!postResponse.startsWith("OK|")) {
                System.out.println("Ошибка отправки капчи: " + postResponse);
                return null;
            }
            String captchaId = postResponse.split("\\|")[1];
            Thread.sleep(GET_DELAY);
            String getUrl = SITE + "res.php?key=" + URLEncoder.encode(API_KEY, "UTF-8") + "&action=get&id=" + URLEncoder.encode(captchaId, "UTF-8");
            String getResponse = sendGet(getUrl);

            if (!getResponse.startsWith("OK|")) {
                System.out.println("Капча ещё не решена: " + getResponse);
                return null;
            }

            return getResponse.split("\\|")[1];

        } catch (Exception e) {
            System.out.println("Ошибка: " + e.getMessage());
            return null;
        }
    }
    
    public static String solveCaptcha(BufferedImage image) {
        try {
            String base64Image = imageToBase64(image);
            String postParams = "key=" + URLEncoder.encode(API_KEY, "UTF-8") + "&method=base64" + "&body=" + URLEncoder.encode(base64Image, "UTF-8");
            String postResponse = sendPost(SITE + "in.php", postParams);
            
            if (!postResponse.startsWith("OK|")) {
                System.out.println("Ошибка отправки капчи: " + postResponse);
                return null;
            }
            String captchaId = postResponse.split("\\|")[1];
            Thread.sleep(GET_DELAY);
            String getUrl = SITE + "res.php?key=" + URLEncoder.encode(API_KEY, "UTF-8") + "&action=get&id=" + URLEncoder.encode(captchaId, "UTF-8");
            String getResponse = sendGet(getUrl);

            if (!getResponse.startsWith("OK|")) {
                System.out.println("Капча ещё не решена: " + getResponse);
                return null;
            }

            return getResponse.split("\\|")[1];

        } catch (Exception e) {
            System.out.println("Ошибка: " + e.getMessage());
            return null;
        }
    }

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        System.out.print("Введите имя файла: ");
        String captchaFile = scanner.nextLine();
        // Решение капчи через отправку файла
        String solvedCaptcha = solveCaptcha(captchaFile);
        System.out.println("Решена капча: " + solvedCaptcha);

        scanner.close();
    }
}

  • API_KEY — ключ доступа к Bare API
  • SITE — адрес API сервера
  • TIMEOUT — максимальное время ожидания HTTP-запроса
  • GET_DELAY — задержка перед получением результата

Функция imageToBase64

Преобразует PNG изображение в Base64 строку для отправки через API.

Из файла

	public static String imageToBase64(String imagePath) throws IOException {
    byte[] imageBytes = Files.readAllBytes(Paths.get(imagePath));
    return Base64.getEncoder().encodeToString(imageBytes);
}

Из BufferedImage

										public static String imageToBase64(BufferedImage image) throws IOException {
    ByteArrayOutputStream byteArrayOutputStream = new ByteArrayOutputStream();
    ImageIO.write(image, "png", byteArrayOutputStream);

    return Base64.getEncoder().encodeToString(
        byteArrayOutputStream.toByteArray()
    );
}

Функция sendPost

Выполняет POST запрос к Bare API.

    SITE + "in.php",
    postParams
);

Отправляемые параметры

										key=API_KEY
method=base64
body=BASE64_IMAGE

Функция sendGet

Выполняет GET запрос для получения результата решения капчи. 

										String getResponse = sendGet(getUrl);

Параметры запроса

										key=API_KEY
action=get
id=CAPTCHA_ID

Функция solveCaptcha

Основная функция решения капчи.

  1. Преобразует изображение в Base64
  2. Отправляет изображение через in.php
  3. Получает ID капчи
  4. Ожидает GET_DELAY
  5. Получает результат через res.php

Решение через файл

										String solvedCaptcha = solveCaptcha("captcha.png");

Решение через BufferedImage

										BufferedImage image = ImageIO.read(new File("captcha.png"));

String solvedCaptcha = solveCaptcha(image);

Примечания

  • При сложных капчах можно увеличить GET_DELAY
  • Пока поддерживаются только PNG формат
